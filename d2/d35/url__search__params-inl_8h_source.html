21inline void url_search_params::initialize(std::string_view input) {
22 if (!input.empty() && input.front() == '?') {
23 input.remove_prefix(1);
24 }
25
26 auto process_key_value = [&](const std::string_view current) {
27 auto equal = current.find('=');
28
29 if (equal == std::string_view::npos) {
30 auto name = std::string(current);
31 std::replace(name.begin(), name.end(), '+', ' ');
32 params.emplace_back(unicode::percent_decode(name, name.find('%')), "");
33 } else {
34 auto name = std::string(current.substr(0, equal));
35 auto value = std::string(current.substr(equal + 1));
36
37 std::replace(name.begin(), name.end(), '+', ' ');
38 std::replace(value.begin(), value.end(), '+', ' ');
39
40 params.emplace_back(unicode::percent_decode(name, name.find('%')),
41 unicode::percent_decode(value, value.find('%')));
42 }
43 };
44
45 while (!input.empty()) {
46 auto ampersand_index = input.find('&');
47
48 if (ampersand_index == std::string_view::npos) {
49 if (!input.empty()) {
50 process_key_value(input);
51 }
52 break;
53 } else if (ampersand_index != 0) {
54 process_key_value(input.substr(0, ampersand_index));
55 }
56
57 input.remove_prefix(ampersand_index + 1);
58 }
59}

110 std::string out{};

129inline void url_search_params::set(const std::string_view key,
130 const std::string_view value) {
131 const auto find = [&key](auto &param) { return param.first == key; };
132
133 auto it = std::find_if(params.begin(), params.end(), find);
134
135 if (it == params.end()) {
136 params.emplace_back(key, value);

138 it->second = value;
